#include <fstream>
#include <vector>

int ver, reb, cnt = 0;
std::vector<std::vector<bool>> tabl_sm;
std::vector<int> toch;

void dfs(int i, int v) {
  
  toch[v] = 1;
  
  for (int j(0); j < ver; j++) {
    
    if (toch[j] == 0 && tabl_sm[v][j] == true) {
      dfs(v, j);
    }
    if (toch[j] == 1 && j != i && j != v && tabl_sm[v][j] == true) {
      cnt++;
    }
    
  }
  
  toch[v] = 2;
  
}

int main() {
  
  std::ifstream fin("input.txt");
  std::ofstream fout("output.txt");
  
	fin >> ver >> reb;
	
	tabl_sm.resize(ver, std::vector<bool>(ver, false));
	toch.resize(ver);
	int a, b;
	
	for (int i(0); i < reb; i++) {
		fin >> a >> b;
		tabl_sm[a-1][b-1] = true;
		tabl_sm[b-1][a-1] = true;
	}
	
	dfs(0, 0);
  
	for (int i(0); i < toch.size(); i++) {
		if (toch[i] == 0 || cnt > 0) {
			fout << "NO";
			return 0;
		}
	}
	
	fout << "YES";
	return 0;
}
