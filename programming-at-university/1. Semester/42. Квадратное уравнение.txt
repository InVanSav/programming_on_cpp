#include <iostream>
#include <fstream>
#include <cmath>
#include <iomanip>

int main() {

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  float a, b, c, D, x_1, x_2;
  fin >> a >> b >> c;
  
  D = pow(b, 2) - 4 * (a * c);
  
  if (a != 0 && D >= 0) {
      x_1 = ((b * -1) + sqrt(D)) / (2 * a);
      x_2 = ((b * -1) - sqrt(D)) / (2 * a);
      fout << x_1 << " " << x_2;
  }
  
  else if (a == 0 && b != 0){
      x_1 = (c / b) * -1;
      fout << x_1 << " " << x_1;
  }
  
  else {
      fout << -1;
    }

  return 0;
}
