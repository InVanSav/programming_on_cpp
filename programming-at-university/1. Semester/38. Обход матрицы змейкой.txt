#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>

using namespace std;

int main(){

  ifstream fin ("input.txt");
  ofstream fout ("output.txt");
  
  int n, p = 1;
  fin >> n;
  int arr[n][n];

  for (int i = 0; i < n; i++){
    for (int j = 0; j < n; j++){
      if (i % 2 == 0){
        arr[i][j] = p++;
      }
      if (i % 2 != 0){
        arr[i][(n-j)-1] = p++;
      }
    }
  }
  for (int i = 0; i < n; i++){
    for (int j = 0; j < n; j++){
      fout << arr[i][j] << " ";
    }
    fout << endl;
  }
  return 0;
}
