#include <iostream>
#include <fstream>
#include <iomanip>

int main() {
  
  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int arr[5], max = 0, min = 1000000;
  long double sum = 0;
  
  for (int i = 0; i < 5; i++) {
    fin >> arr[i];
    if (arr[i] > max) {
      max = arr[i];
    }
    if (arr[i] < min) {
      min = arr[i];
    }
  }
  fout << min << " " << max << std::endl;

  for (int i = 0; i < 5; i++) {
    if (arr[i] != max && arr[i] != min) {
      sum += arr[i];
    }
  }

  sum = sum / 3;

  fout << std::fixed << std::setprecision(2) << sum;
  return 0;
}
