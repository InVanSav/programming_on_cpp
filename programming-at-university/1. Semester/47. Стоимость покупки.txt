#include <iostream>
#include <fstream>
#include <cmath>
#include <iomanip>

int main() {
  
  unsigned int n, a, b, c;
  std::cin >> a >> b >> n;
  
  a *= n;
  b *= n;
  c = b;
  if (b >= 100) {
    b /= 100;
    a += b;
    b = c % 100;
  }
  std::cout << a << " " << b;
  return 0;
}
