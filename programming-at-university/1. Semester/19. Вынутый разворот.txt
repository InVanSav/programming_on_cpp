#include <iostream>
#include <fstream>

using namespace std;

int main ()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");

  unsigned int str_1, str_2;
  fin >> str_1 >> str_2;

  int kol_str;

  if ((str_1 + str_2) && ((str_1 + str_2) % 4 == 1) > kol_str)
  {
  kol_str = ((str_1 + str_2)/4) * 4;
  fout << kol_str;
  }
  else
  {
    fout << 0;
  }
  return 0;
}
