#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>

int main() {

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int a, b, c, arr[10001] = {0};
  fin >> a >> c >> b;
  int start, end, x = 0;

  while (c--) {
    fin >> start >> end;
    arr[start-1]++;
    arr[end]--;
  }
  for (int i = 0; i < 10001; i++) {
    x += arr[i];
    arr[i] = x;
  }
  for (int i = 0; i < b; i++) {
    fin >> x;
    fout << arr[x-1] << std::endl;
  }
  return 0;
}
