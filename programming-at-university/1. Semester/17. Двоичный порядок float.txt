#include <iostream>
using namespace std;
int main()
{
float vesh_ch;

per = per << 1;

if((vesh_ch!=0) && (per==0))
{
  cout << "-126";
} 
else if((per==0)&&(vesh_ch==0)) 
{
  cout << 0;
}
else 
{
  cout << per-127;
}

}
