#include <iostream>
#include <fstream>

int main() {
  
  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int n, count = 0;
  std::cin >> n;
  int arr[n];

  for (int i = 0; i < n; i++) {
    std::cin >> arr[i];
  }
  for (int i = 0; i < n; i++) {
    if ( i - 1 >= 0 && i + 1 < n ) {
      if ( arr[i] > arr[i-1]  &&  arr[i] > arr[i+1] ) {
        count++;
      }
    }
  }
  std::cout << count;
  return 0;
}
