#include <iostream>
using namespace std;

int main()
{
  int Number;
  cin >> Number;
  
  if (Number != 0 && (Number % 3 == 0) && (Number % 5 == 0))
  {
  cout << "FizzBuzz";
  }
  else 
  if(Number != 0 && Number % 5 == 0)
  {
  cout << "Buzz";
  }
  else 
  if (Number != 0 && Number % 3 == 0)
  {
  cout << "Fizz";
  }  
  else 
  {
    cout << endl;
  }
  
  return 0; 
}
