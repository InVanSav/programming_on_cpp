#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>

int main() {

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int sum_kg, n, max = 0, backpack = 0, suitcase = 0;
  fin >> sum_kg >> n;
  int arr[n];

  for (int i = 0; i < n; i++) {
    fin >> arr[i];

    if (arr[i] <= sum_kg && (backpack + arr[i]) <= sum_kg) {
      backpack += arr[i];
    }
    else {
      suitcase += arr[i];
    }
  }

  fout << backpack << " " << suitcase;


  return 0;
}
