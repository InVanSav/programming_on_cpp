#include <iostream>
#include <fstream>

using namespace std;

int main()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");

  unsigned int chok, creamy, kol_vaf_fl;
  fin >> chok >> creamy >> kol_vaf_fl;
  int i = 0, floor_chok = 0, floor_creamy = 0, floor = 0;
  
  while ((kol_vaf_fl <= chok) && (kol_vaf_fl <= creamy))
    {
      chok -= kol_vaf_fl;
      floor_chok += 1;
      creamy -= kol_vaf_fl;
      floor_creamy += 1;
    }
  if (chok >= kol_vaf_fl || creamy >= kol_vaf_fl)
    { 
      floor_chok += 1;
      floor = floor_chok + floor_creamy;
    fout << floor << endl;
    }
  else
  {
    floor = floor_chok + floor_creamy;
  fout << floor << endl;
  }

 return 0;
}
