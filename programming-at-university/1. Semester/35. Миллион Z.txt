#include <iostream>
#include <fstream>

int main(){

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int number_lines, ind_start, ind_finish, count = 0, N = 1000000;
  fin >> number_lines;
  char letter, Z[N];

  for (int i = 0; i < N; i++){
    Z[i] = 'Z';
  }
  for (int i = 0; i < number_lines; i++){
    fin >> letter >> ind_start >> ind_finish;
    for (int j = ind_start-1; j <= ind_finish-1; j++){
      Z[j] = letter;
    }
  }
  letter = (char)65;
  for (int i = 1; i <= 26; i++){
    for (int j = 0; j < N; j++){
      if (Z[j] == letter){
        count++;
        break;
      }
    }
    letter += 1;
  }

  fout << count;

  return 0;
}
