#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>

int main() {

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int n, min = 1000000; 
  fin >> n;
  int arr[n], m, arr_1[n];

  for (int i = 0; i < n; i++) {
    fin >> arr[i];
  }
  
  fin >> m;
  
  for (int i = 0; i < n; i++) {
    arr_1[i] = abs(m - arr[i]); 
  }
  for (int i = 0; i < n; i++) {
    if (arr_1[i] <= min) {
      min = arr_1[i];
    }
  }
  for (int i = 0; i < n; i++) {
    if (arr_1[i] == min) {
      fout << arr[i];
      break;
    }
  }

  return 0;
}
