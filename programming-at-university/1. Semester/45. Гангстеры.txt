#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>
#include <algorithm>

struct Man {

  int time;
  int money;
  int width;
  Man():
    time(0), 
    money(0),
    width(0)
  {}
};

int main() {
  
  int n;
  scanf("%d %*d %*d", &n);
  std::vector<Man> a(n);

  for (int i = 0; i < n; i++) {
    scanf("%d", &a[i].time);
  }
  for (int i = 0; i < n; i++) {
    scanf("%d", &a[i].money);
  }
  for (int i = 0; i < n; i++) {
    scanf("%d", &a[i].width);
  }
  std::sort(a.begin(), a.end(), [](const Man &left, const Man &right) {
    return left.time < right.time;
  });

  std::vector<int> max_sum(n, 0);
  for (int last = 0; last < n; last++) {
    if (a[last].width > a[last].time) {
      continue;
    }
    else {
      max_sum[last] = a[last].money;
      for (int prev = 0; prev < last; prev++) {
        if (std::abs(a[prev].width - a[last].width) > a[last].time - a[prev].time) {
          continue;
        }
        else {
          max_sum[last] = std::max(max_sum[last], max_sum[prev] + a[last].money);
        }
      }
    }
  }

  int max = 0;
  for (int i = 0; i < n; i++) {
    max = std::max(max, max_sum[i]);
  }
  printf("%d", max);
  
  return 0;
}
