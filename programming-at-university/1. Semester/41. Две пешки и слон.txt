#include <iostream>
#include <fstream>

int main() {

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int x_1, x_2, y_1, y_2, x_add = 1, y_add = 1;
  int elephant_1 = -8, elephant_2 = -8;
  fin >> x_1 >> y_1 >> x_2 >> y_2;

  for (int i = 0; i < 64; i++) {
    if ( abs((y_add - y_1)) == abs((x_add - x_1)) ) {
      if ( abs((x_add - x_2)) == abs((y_add - y_2)) ) {
        if ( ( abs((x_2 - x_1)) > 1 || abs((y_2 - y_1)) > 1 ) ) {
          if ( (x_2 > x_add && x_1 < x_add) || (y_2 > y_add && y_1 < y_add)  ||  (x_2 < x_add && x_1 > x_add) || (y_2 < y_add && y_1 > y_add) ) {
            elephant_1 = x_add; 
            elephant_2 = y_add;
          }
        }
      }
    }
    if ( x_add != 8 ) {
      x_add++;
    }
    else {
      x_add = 1;
      y_add++;
    }
  }
  
  if ( elephant_1 == -8 ) {
    fout << 0;
  }
  else {
    fout << elephant_1 << " " << elephant_2;
  }

  return 0;
}
