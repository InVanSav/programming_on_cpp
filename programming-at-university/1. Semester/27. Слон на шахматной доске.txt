#include <iostream>
#include <fstream>

using namespace std;

int main()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");
  
  unsigned short n, gor, ver;
  fin >> n >> gor >> ver;
  char shah_dos[n][n];
  
  gor--;
  ver--;

  for (int j = 0; j < n; j++)
  {    
    for (int i = 0; i < n; i++)
    {
      if ((i % 2 == 0 && j % 2 == 0) || (i % 2 == 1 && j % 2 == 1))
      {
        shah_dos[i][j] = '#';
      }
      else if ((i % 2 == 0 && j % 2 == 1) || (i % 2 == 1 && j % 2 == 0))
      {
        shah_dos[i][j] = '.';
      }
    }
  } 
  for (int i = 0; i < n; i++)
  {
    if ((gor - i >= 0) && (ver - i >= 0))
    {
      shah_dos[gor-i][ver-i] = '*';
    }
    if ((gor + i < n) && (ver - i >= 0))
    {
      shah_dos[gor+i][ver-i] = '*';
    }
    if ((gor - i >= 0) && (ver + i < n))
    {
      shah_dos[gor-i][ver+i] = '*';
    }
    if ((gor + i < n) && (ver + i < n))
    {
      shah_dos[gor+i][ver+i] = '*';
    }
  }
  for (int j = 0; j < n; j++)
  {    
    for (int i = 0; i < n; i++)
    {
      shah_dos[gor][ver] = 'X';
      fout << shah_dos[i][j];
    }
  fout << endl;
  }
  return 0;
}
