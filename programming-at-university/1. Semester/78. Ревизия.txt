#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>

int main() {

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");

  int c, b, a, l;
  int res[2];

  fin >> c;
  fin >> b >> a;

  if (b < a) {
    res[0] = b;
    res[1] = a;
  }
  else {
    res[0] = a;
    res[1] = b;
  }

  for (int i = 2; i < c; i++) {
    fin >> l;
    if (l > res[1]) {
      continue;
    }
    if (l < res[0]) {
      res[1] = res[0];
      res[0] = l;
    }
    else {
      res[1] = l;
    }
  }
  fout << res[0] << " " << res[1];
  return 0;
}
