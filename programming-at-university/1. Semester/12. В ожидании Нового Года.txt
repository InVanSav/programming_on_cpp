#include <iostream>
#include <fstream>

using namespace std;

int main()
{
ifstream fin ("input.txt");
ofstream fout ("output.txt");

int ost_h, ost_m, Hour, Minuts;

fin >> Hour >> Minuts;

ost_h = 23 - Hour;
ost_m = 60 - Minuts;

if (Hour == 23)
{
  ost_h = 0;
  ost_m = 60 - Minuts;
}
if (Hour == 22) 
{
  ost_h = 23 - Hour;
  ost_m = 60 - Minuts;
}
if (Minuts == 0)
{
  ost_h = 24 - Hour;
  ost_m = 0;
}

fout << ost_h << " " << ost_m;

return 0;
}
