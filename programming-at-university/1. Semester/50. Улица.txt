#include <iostream>
#include <fstream>
#include <cmath>
#include <iomanip>

int main() {
  
  unsigned int a, b, count = 0;
  std::cin >> a >> b;
  
  if (a < b) {
    (a % 2 == 0) ? a : a++;
    while (a < b) {
      a += 2;
      count++;
    }
  }
  else if (a > b) {
    (b % 2 == 0) ? b : b++;
    while (a > b) {
      b += 2;
      count++;
    }
  }
  
  std::cout << count;
  return 0;
}
