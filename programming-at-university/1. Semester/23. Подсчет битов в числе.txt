#include <iostream>
#include <fstream>

using namespace std;

int main()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");

size_t num_zeroes = 0, chislo, operation;
fin >> chislo;

  for(size_t i = 0; i < sizeof(char) *chislo; chislo /= 2){
    operation = chislo % 2;

      if (operation == 1){
        ++num_zeroes;     
      }
  }

  fout << num_zeroes << endl;
 return 0;
}
