#include <iostream>
#include <fstream>

int main()
{
  std::ifstream fin ("negative.in");
  std::ofstream fout ("negative.out");
  
  unsigned int n, m, count = 0;
  fin >> n >> m;
  char b_w[n][m], w_b[n][m];

  for (int i = 0; i < n; i++){
    for (int j = 0; j < m; j++){
      fin >> b_w[i][j];
      if (b_w[i][j] == 'B'){
        b_w[i][j] = 'W';
      }
      else if (b_w[i][j] == 'W'){
        b_w[i][j] = 'B';
      }
    }
  }
  for (int i = 0; i < n; i++){
    for (int j = 0; j < m; j++){
      fin >> w_b[i][j];
      if (w_b[i][j] != b_w[i][j]){
        count += 1;
      }
    }
  }
  fout << count;
  

  return 0;
}
