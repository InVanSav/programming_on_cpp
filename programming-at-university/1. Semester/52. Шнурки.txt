#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>

int main() {

  std::ifstream fin("input.txt");
  std::ofstream fout("output.txt");

  int l, n, b, a, lace = 0;
  fin >> a >> b >> l >> n;

  lace += 2 * l;
  lace += 2 * b * (n - 1);
  lace += 2 * a * (n - 1) + a;

  fout << lace;

  return 0;
}
