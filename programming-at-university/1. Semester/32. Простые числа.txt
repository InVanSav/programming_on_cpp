#include <iostream>
#include <fstream>
#include <cmath>

using namespace std;

int main()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");

  unsigned int number;
  fin >> number;
  int digit = 2, i = 0;

  while (i < number)
  {
    bool value = true;
    for (int j = 2; j <= ceil(sqrt(digit)); j++) 
    {
      value = value && (digit % j != 0 || digit / 2 == 1); 
    }
    if (value)
    {
      fout << digit << " ";
      i++;
    }
    digit++;
  }
  return 0;
}
