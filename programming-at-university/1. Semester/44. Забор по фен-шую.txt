#include <iostream>
#include <fstream>
#include <cmath>
#include <iomanip>

int main() {

  std::ifstream fin ("input.txt");
  std::ofstream fout ("output.txt");
  
  int n;
  fin >> n;
  unsigned int arr[n], arr_1[n], count_1 = 0, count_2 = 0;

  for (int i = 0; i < n; i++) {
    fin >> arr[i];
    arr_1[i] = arr[i];
  }

  for (int i = 0; i < n; i++) {

    if (i % 2 == 0) {

      if (arr[i-1] >= arr[i] && i - 1 >= 0) {
        arr[i] = arr[i-1] + 1;
      }
      if (arr[i+1] >= arr[i] && i + 1 < n) {
        arr[i] = arr[i+1] + 1;
      }
    }
  }
  
  for (int i = 0; i < n; i++) {

    if (i % 2 != 0) {

      if (arr_1[i-1] >= arr_1[i] && i - 1 >= 0) {
        arr_1[i] = arr_1[i-1] + 1;
      }
      if (arr_1[i+1] >= arr_1[i] && i + 1 < n) {
        arr_1[i] = arr_1[i+1] + 1;
      }
    }
  }
  
  for (int i = 0; i < n; i++) {
    count_1 += arr[i];
    count_2 += arr_1[i];
  }
  
  if (count_2 >= count_1) {
    for (int i = 0; i < n; i++) {
      fout << arr[i] << " ";
    }
  } 
  else if (count_2 <= count_1) {
    for (int i = 0; i < n; i++) {
      fout << arr_1[i] << " ";
    }
  }
   

  return 0;
}
