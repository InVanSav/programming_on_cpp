#include <iostream>
#include <fstream>

using namespace std;

int main()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");
  
  int amount;
  fin >> amount;
  unsigned int score[amount], sum = 0;
  char pass_fail[amount];
  

  for (int i = 0; i < amount; i++)
  {
    fin >> score[i];
  }
  for (int i = 0; i < amount; i++)
  {
    fin >> pass_fail[i];
  }
  for (int j = 0; j < amount; j++)
  {
    if (pass_fail[j] == (char)43)
      {
        sum += score[j];
      }
  }
  fout << sum << endl;
  
  return 0;
}
