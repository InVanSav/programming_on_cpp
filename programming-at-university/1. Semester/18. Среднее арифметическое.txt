#include <stdio.h>

int main()
{
  FILE* fin = fopen ("input.txt", "r");
  FILE* fout = fopen ("output.txt", "w");

  double n, chislo;  
  fscanf (fin, "%lf", &n);
  
  int i = 0;
  double sr_ar = 0, sum = 0;

  while (i < n)
  {
    fscanf (fin, "%lf", &chislo);
    sum += chislo;
    i += 1;
  }
  sr_ar = sum/n;
fprintf (fout, "%lf", sr_ar);
return 0;
}

#include <iostream>
#include <fstream>
#include <iomanip>

using namespace std;

int main()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");

  int n;  
  fin >> n;

  int chislo[n], i = 0;
  double sr_ar = 0.0000, sum = 0.0000;

  while (i < n)
  {
    fin >> chislo[i];
    sum += chislo[i];
    sr_ar = sum/n;
    i += 1;
  }
 
  fout << fixed << setprecision(10) << sr_ar << endl;
  
return 0;
}
