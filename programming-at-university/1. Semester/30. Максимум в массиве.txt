#include <iostream>
#include <fstream>

using namespace std;

int main()
{
  ifstream fin ("input.txt");
  ofstream fout ("output.txt");
  
  int n;
  fin >> n;
  int arr[n], max;

  for (int i = 0; i < n; i++)
  {
    fin >> arr[i];
  } 
  max = arr[0];
  for (int i = 1; i < n; i++)
  {
    if (arr[i] > max)
    {
      max = arr[i];
    }
  }
  fout << max;
  return 0;
}
