#include <iostream>
#include <fstream>
using namespace std;

int main()
{
  ifstream fin ("input.txt");//файл для чтения
  ofstream fout ("output.txt");//файл для записи
  
  unsigned int lengths, time_1, time_2, Time;
  fin >> lengths >> time_1 >> time_2 >> Time; //записали значения в файл
  int kol_circles, ost_time, metr;
      kol_circles = (Time * 60)/(time_1 + time_2);//кол-во кругов
      metr = kol_circles * lengths;//сколько проплыла за эти круги
      ost_time = (Time*60) % (time_1+time_2);//остаток времени

  if (ost_time > time_1)
  {
    fout << (metr += lengths);
  }
  else
  {
    fout << (metr += (ost_time * lengths)/time_1);
  }
  return 0;
}
