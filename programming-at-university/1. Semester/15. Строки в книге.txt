#include <iostream>
#include <fstream>

using namespace std;

int main()
{
  ifstream fin ("a.in");
  ofstream fout ("a.out");

unsigned int kol_strok, nomer_stroki;
fin >> kol_strok >> nomer_stroki;

int str = 0;

if (nomer_stroki < kol_strok)
{
    str = 1;
fout << str << " " << nomer_stroki;
}
else
{
    while (nomer_stroki > kol_strok) 
    {
      nomer_stroki -= kol_strok;
      str += 1;
    }
  str += 1;
fout << str << " " << nomer_stroki;
}

return 0;
}
