#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <sstream>
#include <deque>
#include <cmath>

using namespace std;

int main() {

  ifstream fin ("input.txt");
  ofstream fout ("output.txt");

  int flashlight[101];

  for (int i = 0; i <= 100; i++) {
    flashlight[i] = 0;
  }

  int x, x_1, x_2, y, n;
  fin >> n;

  for (int i = 1; i <= n; i++) {
    fin >> x >> y;
    x_1 = x - y + 1;
    x_2 = x + y;
    (x_1 < 0) ? x_1 = 0 : 0;
    (x_2 > 100) ? x_2 = 100 : 0;
    for (int j = x_1; j <= x_2; j++) {
      flashlight[j]++;
    }
  }
  int max = 0;
  for (int i = 0; i <= 100; i++) {
    if (flashlight[i] > max){
      max = flashlight[i];
    }
  }
  fout << max;
  return 0;
}
